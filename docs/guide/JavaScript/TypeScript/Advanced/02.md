# 装饰器（Decorators）

## 1. 装饰器概述

1. 装饰器是一种特殊类型的声明，能够附加到类、方法、属性或参数上，在不修改原始代码的情况下扩展其行为。
2. TypeScript 5.0+ 支持 **ECMAScript 标准装饰器**（Stage 3），与旧版实验性装饰器不兼容。
3. 装饰器本质上是一个**函数**，它接收被装饰目标的信息并返回修改后的结果。

```typescript
// 装饰器是一个函数，第一个参数是被装饰的目标
function log(target: any, context: ClassMemberDecoratorContext) {
  console.log(`装饰: ${String(context.name)}`)
}
```

## 2. 类装饰器

类装饰器用于修改或替换类的定义：

```typescript
// 给类添加额外的静态属性
function addVersion<T extends new (...args: any[]) => any>(target: T) {
  return class extends target {
    version = '1.0.0'
  }
}

@addVersion
class App {
  name = 'MyApp'
}

const app = new App()
console.log((app as any).version)  // '1.0.0'
```

### 阻止类被继承

```typescript
function sealed<T extends new (...args: any[]) => any>(target: T) {
  Object.seal(target)
  Object.seal(target.prototype)
  return target
}

@sealed
class LockedClass {
  x = 1
}

// 试图扩展 sealed 的类会出错
// class Child extends LockedClass {} // ❌ (严格模式下)
```

## 3. 方法装饰器

方法装饰器可以替换方法，或添加前置/后置逻辑：

```typescript
// 记录方法调用日志
function logged(
  target: Function,
  context: ClassMethodDecoratorContext
) {
  const methodName = String(context.name)

  return function (this: any, ...args: any[]) {
    console.log(`调用 ${methodName}(${args.join(', ')})`)
    const start = Date.now()
    const result = target.apply(this, args)
    const elapsed = Date.now() - start
    console.log(`${methodName} 执行完毕，耗时 ${elapsed}ms`)
    return result
  }
}

class Calculator {
  @logged
  add(a: number, b: number): number {
    return a + b
  }

  @logged
  multiply(a: number, b: number): number {
    return a * b
  }
}

const calc = new Calculator()
calc.add(1, 2)
// 调用 add(1, 2)
// add 执行完毕，耗时 0ms

calc.multiply(3, 4)
// 调用 multiply(3, 4)
// multiply 执行完毕，耗时 0ms
```

### 方法缓存（Memoization）

```typescript
function memoize(
  target: any,
  context: ClassMethodDecoratorContext
) {
  const cache = new Map<string, any>()

  return function (this: any, ...args: any[]) {
    const key = JSON.stringify(args)
    if (cache.has(key)) {
      console.log('从缓存返回')
      return cache.get(key)
    }
    const result = target.apply(this, args)
    cache.set(key, result)
    return result
  }
}

class MathOps {
  @memoize
  fibonacci(n: number): number {
    if (n <= 1) return n
    return this.fibonacci(n - 1) + this.fibonacci(n - 2)
  }
}

const math = new MathOps()
console.log(math.fibonacci(40))  // 第一次计算
console.log(math.fibonacci(40))  // 从缓存返回
```

## 4. 属性装饰器

```typescript
// 将属性值转换为特定格式
function toUpperCase(
  target: undefined,
  context: ClassFieldDecoratorContext
) {
  return function (this: any, value: string) {
    return value.toUpperCase()
  }
}

function trim(
  target: undefined,
  context: ClassFieldDecoratorContext
) {
  return function (this: any, value: string) {
    return value.trim()
  }
}

class UserForm {
  @toUpperCase
  @trim
  name: string = '  alice  '

  @trim
  email: string = '  alice@example.com  '
}

const form = new UserForm()
console.log(form.name)  // 'ALICE'
console.log(form.email) // 'alice@example.com'
```

## 5. 访问器装饰器（Getter/Setter）

```typescript
function validate(
  target: any,
  context: ClassSetterDecoratorContext
) {
  return function (this: any, value: number) {
    if (value < 0) {
      throw new Error(`${String(context.name)} 不能为负数`)
    }
    target.call(this, value)
  }
}

class Product {
  private _price: number = 0

  get price(): number {
    return this._price
  }

  @validate
  set price(value: number) {
    this._price = value
  }
}

const p = new Product()
p.price = 100  // ✅
// p.price = -10  // ❌ 抛出错误
```

## 6. 装饰器工厂

装饰器工厂是返回装饰器的函数，用于接收额外参数：

```typescript
// 装饰器工厂 —— 接收参数
function delay(ms: number) {
  // 返回的是真正的装饰器
  return function (
    target: Function,
    context: ClassMethodDecoratorContext
  ) {
    return async function (this: any, ...args: any[]) {
      await new Promise(resolve => setTimeout(resolve, ms))
      return target.apply(this, args)
    }
  }
}

class API {
  @delay(1000)  // 延迟 1 秒执行
  async fetchData() {
    console.log('获取数据...')
    return { data: 'success' }
  }

  @delay(200)   // 延迟 200 毫秒
  async quickCheck() {
    console.log('快速检查...')
    return true
  }
}
```

## 7. 参数装饰器

参数装饰器可以获取参数的元数据：

```typescript
function required(
  target: undefined,
  context: ClassMethodDecoratorContext<Function>,
  parameterIndex: number
) {
  // 在元数据中记录必须的参数
  console.log(`参数 #${parameterIndex} 被标记为 required`)
}

class UserService {
  createUser(
    @required name: string,
    @required email: string,
    age?: number
  ) {
    return { name, email, age }
  }
}
```

## 8. 综合实战案例

### 案例一：防抖装饰器

```typescript
function debounce(delay: number) {
  return function (
    target: Function,
    context: ClassMethodDecoratorContext
  ) {
    let timer: ReturnType<typeof setTimeout> | null = null

    return function (this: any, ...args: any[]) {
      if (timer) clearTimeout(timer)
      timer = setTimeout(() => {
        target.apply(this, args)
        timer = null
      }, delay)
    }
  }
}

class SearchBox {
  @debounce(300)
  onSearch(keyword: string) {
    console.log('搜索:', keyword)
    // 实际请求 API
  }
}

const box = new SearchBox()
box.onSearch('t')     // 300ms 内连续调用
box.onSearch('ty')    // 只有最后一次
box.onSearch('typ')   // '搜索: typ' 实际执行
```

### 案例二：权限检查装饰器

```typescript
const ROLES_KEY = Symbol('roles')

function RequireRole(role: string) {
  return function (
    target: Function,
    context: ClassMethodDecoratorContext
  ) {
    return function (this: any, ...args: any[]) {
      // 假设从某处获取当前用户
      const currentUser = { roles: ['admin', 'editor'] }

      if (!currentUser.roles.includes(role)) {
        throw new Error(`需要 ${role} 权限，当前用户无此权限`)
      }
      return target.apply(this, args)
    }
  }
}

class AdminController {
  @RequireRole('admin')
  deleteUser(userId: number) {
    console.log(`删除用户 ${userId}`)
  }

  @RequireRole('editor')
  editArticle(articleId: number) {
    console.log(`编辑文章 ${articleId}`)
  }
}
```

### 案例三：自动绑定 this

```typescript
function autobind(
  target: Function,
  context: ClassMethodDecoratorContext
) {
  return function (this: any, ...args: any[]) {
    return target.apply(this, args)
  }
}

class Button {
  label: string

  constructor(label: string) {
    this.label = label
  }

  @autobind
  handleClick(event: Event) {
    // 即使作为回调传递，this 始终指向 Button 实例
    console.log(`按钮 "${this.label}" 被点击`)
  }
}
```

## 9. 装饰器执行顺序

当同时使用多种装饰器时，执行顺序如下：

1. **参数装饰器** → **方法/属性/访问器装饰器** → **类装饰器**
2. 同类型的装饰器**从下到上**依次执行（靠近目标的内层装饰器先执行）

```typescript
function first(msg: string) {
  return function (...args: any[]) {
    console.log(`[first] ${msg}`)
  }
}

function second(msg: string) {
  return function (...args: any[]) {
    console.log(`[second] ${msg}`)
  }
}

@first('类装饰器')
class Example {
  @second('属性装饰器')
  name: string = 'test'

  @first('方法装饰器')
  @second('方法装饰器 2')
  greet() {}

  constructor(@second('参数装饰器') x: number) {}
}

// 输出顺序：
// [second] 参数装饰器
// [second] 属性装饰器
// [second] 方法装饰器 2
// [first] 方法装饰器
// [first] 类装饰器
```

## 10. 装饰器最佳实践

1. **TypeScript 5.0+ 使用标准装饰器**：无需 `experimentalDecorators` 标志，语法更接近 JS 标准。
2. **优先用装饰器工厂**：可参数化的装饰器更灵活。
3. **装饰器保持单一职责**：一个装饰器只做一件事（日志、缓存、权限等独立）。
4. **注意装饰器开销**：每个装饰器都会增加函数调用的开销，避免在性能敏感的热路径上堆叠过多装饰器。
5. **装饰器是可组合的**：可以叠加多个装饰器，执行顺序从下到上。
