# 泛型（Generics）深入

## 1. 为什么需要泛型

在不使用泛型时，你可能这样写：

```typescript
// 方式一：any —— 丢失类型信息
function identity1(arg: any): any {
  return arg
}
const r1 = identity1('hello')  // r1 类型为 any

// 方式二：重载 —— 每增加一种类型就要多写一份
function identity2(arg: number): number
function identity2(arg: string): string
function identity2(arg: any): any {
  return arg
}
```

泛型让我们**在定义时用类型变量占位，调用时才确定具体类型**：

```typescript
function identity<Type>(arg: Type): Type {
  return arg
}

const r2 = identity('hello')  // r2 类型为 'hello'（字面量）
const r3 = identity(42)       // r3 类型为 42
const r4 = identity<string>('world')  // 也可以显式指定
```

## 2. 泛型约束（extends）

### 基础约束

```typescript
// 约束 Type 必须有 length 属性
function getLength<T extends { length: number }>(arg: T): number {
  return arg.length
}

getLength('hello')     // OK，string 有 length
getLength([1, 2, 3])   // OK，数组有 length
getLength({ length: 5, name: 'x' })  // OK
// getLength(123)      // ❌ number 没有 length
```

### 多类型参数约束

```typescript
// K 必须是 T 的键名
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key]
}

const person = { name: 'yzb', age: 18 }
getProperty(person, 'name')  // ✅ 返回 string
getProperty(person, 'age')   // ✅ 返回 number
// getProperty(person, 'email') // ❌ 'email' 不是 person 的键
```

### 类型参数之间的约束

```typescript
// 用一个类型参数约束另一个
function copyFields<T extends U, U>(target: T, source: U): T {
  for (const id in source) {
    target[id] = source[id] as any
  }
  return target
}

const x = { a: 1, b: 2, c: 3, d: 4 }
copyFields(x, { b: 10, d: 20 })  // { a: 1, b: 10, c: 3, d: 20 }
```

## 3. 泛型类

```typescript
class Stack<T> {
  private items: T[] = []

  push(item: T): void {
    this.items.push(item)
  }

  pop(): T | undefined {
    return this.items.pop()
  }

  peek(): T | undefined {
    return this.items[this.items.length - 1]
  }

  get size(): number {
    return this.items.length
  }
}

const numStack = new Stack<number>()
numStack.push(1)
numStack.push(2)
console.log(numStack.pop())  // 2

const strStack = new Stack<string>()
strStack.push('hello')
console.log(strStack.peek()) // 'hello'
```

## 4. 泛型接口

```typescript
// 泛型接口定义
interface Result<T> {
  code: number
  message: string
  data: T
}

// 使用
const userResult: Result<{ name: string; age: number }> = {
  code: 0,
  message: 'success',
  data: { name: 'yzb', age: 18 }
}

const listResult: Result<string[]> = {
  code: 0,
  message: 'success',
  data: ['a', 'b', 'c']
}

// 泛型函数接口
interface GenericFn {
  <T>(arg: T): T
}

const identity: GenericFn = <T>(arg: T): T => arg
```

## 5. 泛型默认值

```typescript

interface Config<T = string> {
  value: T
  onChange: (newValue: T) => void

const nameConfig: Config = {
  value: 'yzb',
  onChange: (val) => console.log(val)  // val 为 string

  value: 18,
  onChange: (val) => console.log(val)  // val 为 number
```

## 6. 泛型数组与元组

```typescript
// 泛型数组
function first<T>(arr: T[]): T | undefined {
  return arr[0]
}

// 泛型元组 —— 交换两个元素
function swap<T, U>(tuple: [T, U]): [U, T] {
  return [tuple[1], tuple[0]]
}

const swapped = swap(['hello', 42])
// swapped: [number, string]
console.log(swapped) // [42, 'hello']
```

## 7. 条件类型与 infer

### 基础条件类型

```typescript
// 条件类型语法: T extends U ? X : Y
type IsString<T> = T extends string ? 'yes' : 'no'

type A = IsString<string>  // 'yes'
type B = IsString<number>  // 'no'
type C = IsString<string | number>  // 'yes' | 'no' —— 分配律
```

### infer 关键字

`infer` 用于在条件类型中**推断**类型变量：

```typescript
// 提取数组元素类型
type ArrayItem<T> = T extends (infer U)[] ? U : never

type Item1 = ArrayItem<string[]>        // string
type Item2 = ArrayItem<number[]>        // number
type Item3 = ArrayItem<boolean>         // never

// 提取函数返回值类型
type ReturnType2<T> = T extends (...args: any[]) => infer R ? R : never

type R1 = ReturnType2<() => string>     // string
type R2 = ReturnType2<(x: number) => void>  // void

// 提取 Promise 包裹的类型
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T

type P1 = UnwrapPromise<Promise<string>>   // string
type P2 = UnwrapPromise<number>            // number
```

## 8. 映射类型

映射类型基于已有类型创建新类型，逐个转换每个属性：

```typescript
// 将所有属性变为只读
type MyReadonly<T> = {
  readonly [K in keyof T]: T[K]
}

interface User {
  name: string
  age: number
}

type ReadonlyUser = MyReadonly<User>
// { readonly name: string; readonly age: number }

// 将所有属性变为可选
type MyPartial<T> = {
  [K in keyof T]?: T[K]
}

type PartialUser = MyPartial<User>
// { name?: string; age?: number }
```

### 映射修饰符

```typescript
// 去除 readonly 和 可选
type Mutable<T> = {
  -readonly [K in keyof T]: T[K]  // -readonly 移除只读
}

type Required2<T> = {
  [K in keyof T]-?: T[K]          // -? 移除可选
}

interface Config {
  readonly host: string
  port?: number
}

type MutableConfig = Mutable<Config>
// { host: string; port?: number }

type RequiredConfig = Required2<Config>
// { readonly host: string; port: number }
```

### 通过 as 重映射键名

```typescript
// 给所有属性名加上 get 前缀
type Getters<T> = {
  [K in keyof T as `get${Capitalize<string & K>}`]: () => T[K]
}

interface Person {
  name: string
  age: number
}

type PersonGetters = Getters<Person>
// { getName: () => string; getAge: () => number }

// 按条件过滤属性
type PickByType<T, U> = {
  [K in keyof T as T[K] extends U ? K : never]: T[K]
}

interface Mixed {
  name: string
  age: number
  isActive: boolean
  hobbies: string[]
}

type OnlyStrings = PickByType<Mixed, string>
// { name: string }
```

## 9. 模板字面量类型

```typescript
type Color = 'red' | 'green' | 'blue'
type Size = 'small' | 'medium' | 'large'

// 生成所有组合
type ColorSize = `${Color}-${Size}`
// 'red-small' | 'red-medium' | 'red-large' |
// 'green-small' | 'green-medium' | 'green-large' |
// 'blue-small' | 'blue-medium' | 'blue-large'

// 实用案例 —— 事件监听类型
type EventName<T extends string> = `on${Capitalize<T>}`
// EventName<'click'> → 'onClick'
// EventName<'change'> → 'onChange'
```

## 10. 泛型实战案例

### 案例一：通用 API 响应包装

```typescript
interface ApiResponse<T> {
  code: number
  message: string
  data: T
}

interface PageResult<T> {
  list: T[]
  total: number
  page: number
  pageSize: number
}

// 分页查询用户
type UserListResponse = ApiResponse<PageResult<{
  id: number
  name: string
  email: string
}>>

// 实际数据格式：
// { code: 0, message: 'ok', data: { list: [...], total: 100, page: 1, pageSize: 20 } }
```

### 案例二：事件总线

```typescript
// 通过映射确保 emit 和 on 的类型一一对应
class EventBus<Events extends Record<string, any[]>> {
  private handlers = new Map<string, Set<Function>>()

  on<K extends keyof Events>(
    event: K,
    handler: (...args: Events[K]) => void
  ): void {
    if (!this.handlers.has(event as string)) {
      this.handlers.set(event as string, new Set())
    }
    this.handlers.get(event as string)!.add(handler)
  }

  emit<K extends keyof Events>(event: K, ...args: Events[K]): void {
    this.handlers.get(event as string)?.forEach(fn => fn(...args))
  }

  off<K extends keyof Events>(
    event: K,
    handler: (...args: Events[K]) => void
  ): void {
    this.handlers.get(event as string)?.delete(handler)
  }
}

// 使用
type AppEvents = {
  login: [userId: number, timestamp: Date]
  logout: [userId: number]
  message: [content: string, from: string]
}

const bus = new EventBus<AppEvents>()

bus.on('login', (userId, timestamp) => {
  console.log(`用户 ${userId} 在 ${timestamp} 登录`)
})
// ✅ 类型安全：参数自动推断

bus.emit('login', 1, new Date())
// bus.emit('login', 'wrong')  // ❌ 类型错误
```

## 11. 泛型最佳实践

1. **用有意义的名称**：`T` 是通用泛型名，特定场景用有含义的名字（`TItem`、`TData`、`TKey`）。
2. **优先让 TypeScript 推断**：大多数泛型调用不需要手动指定类型参数。
3. **合理使用约束**：用 `extends` 限制泛型可接受的类型范围。
4. **泛型默认值提高易用性**：为常用场景提供默认类型。
5. **避免过度泛型化**：如果只有一两处使用且类型简单，直接写具体类型更清晰。
6. **条件类型 + infer 用于类型体操**：理解 `infer` 是掌握高级类型的关键。
